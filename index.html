let currentStep = 1;

const totalSteps = 4;


function goStep(n) {

  if (n > currentStep && !validateStep(currentStep)) return;

  document.getElementById('step' + currentStep).classList.remove('active');

  currentStep = n;

  document.getElementById('step' + currentStep).classList.add('active');

  updateProgress();

  window.scrollTo({ top: 0, behavior: 'smooth' });

}


function updateProgress() {

  const pct = ((currentStep - 1) / (totalSteps - 1)) * 100;

  document.getElementById('progressFill').style.width = pct + '%';

  for (let i = 1; i <= totalSteps; i++) {

    const dot = document.getElementById('dot' + i);

    const lbl = document.getElementById('lbl' + i);

    dot.classList.remove('active', 'done');

    lbl.classList.remove('active', 'done');

    if (i < currentStep) { dot.classList.add('done'); dot.innerHTML = '✓'; lbl.classList.add('done'); }

    else if (i === currentStep) { dot.classList.add('active'); dot.innerHTML = i; lbl.classList.add('active'); }

    else { dot.innerHTML = i; }

  }

}

function showErr(id, show) {

  const el = document.getElementById('err-' + id);

  if (el) { el.classList.toggle('show', show); }

  const input = document.getElementById(id);

  if (input) { input.classList.toggle('error', show); }

}


function validateStep(step) {

  let valid = true;

  if (step === 1) {

    ['nama', 'tanggal', 'noHP'].forEach(f => {

      const val = document.getElementById(f)?.value.trim();

      if (!val) { showErr(f, true); valid = false; }

      else showErr(f, false);

    });

    const divisi = document.querySelector('input[name="divisi"]:checked');

    if (!divisi) { document.getElementById('err-divisi').classList.add('show'); valid = false; }

    else document.getElementById('err-divisi').classList.remove('show');

  }

  if (step === 2) {

    if (!document.getElementById('jenisKategori').value) { showErr('jenis', true); valid = false; }

    else showErr('jenis', false);

    const nom = document.getElementById('nominal').value;

    if (!nom || parseFloat(nom) <= 0) { showErr('nominal', true); valid = false; }

    else showErr('nominal', false);

    if (!document.getElementById('tujuan').value.trim()) { showErr('tujuan', true); valid = false; }

    else showErr('tujuan', false);

  }

  if (step === 3) {

    if (uploadedFiles.length === 0) {

      document.getElementById('err-file').classList.add('show'); valid = false;

    } else document.getElementById('err-file').classList.remove('show');

  }

  if (step === 4) {

    if (!sigHasContent()) {

      document.getElementById('err-sig').classList.add('show'); valid = false;

    } else document.getElementById('err-sig').classList.remove('show');

  }

  return valid;

}

let uploadedFiles = [];

const previewGrid = document.getElementById('previewGrid');


function handleFiles(files) {

  Array.from(files).forEach(file => {

    if (uploadedFiles.length >= 5) { showToast('❌ Maks. 5 file'); return; }

    if (file.size > 5 * 1024 * 1024) { showToast('❌ File ' + file.name + ' terlalu besar (maks 5MB)'); return; }

    uploadedFiles.push(file);

    const reader = new FileReader();

    reader.onload = e => addPreview(e.target.result, uploadedFiles.length - 1);

    reader.readAsDataURL(file);

  });

  document.getElementById('err-file').classList.remove('show');

}


function addPreview(src, idx) {

  const item = document.createElement('div');

  item.className = 'preview-item'; item.id = 'prev-' + idx;

  item.innerHTML = `<img src="${src}">

    <button class="remove-btn" onclick="removeFile(${idx})">

      <svg viewBox="0 0 24 24"><line x1="18" y1="6" x2="6" y2="18"/><line x1="6" y1="6" x2="18" y2="18"/></svg>

    </button>`;

  previewGrid.appendChild(item);

}


function removeFile(idx) {

  uploadedFiles.splice(idx, 1);

  previewGrid.innerHTML = '';

  uploadedFiles.forEach((f, i) => {

    const reader = new FileReader();

    reader.onload = e => addPreview(e.target.result, i);

    reader.readAsDataURL(f);

  });

}

function getPos(e) {

  const r = canvas.getBoundingClientRect();

  const src = e.touches ? e.touches[0] : e;

  return { x: src.clientX - r.left, y: src.clientY - r.top };

}


canvas.addEventListener('mousedown', e => { drawing = true; const p = getPos(e); ctx.beginPath(); ctx.moveTo(p.x, p.y); });

canvas.addEventListener('mousemove', e => { if (!drawing) return; const p = getPos(e); ctx.lineWidth = 2; ctx.lineCap = 'round'; ctx.strokeStyle = '#0A1628'; ctx.lineTo(p.x, p.y); ctx.stroke(); hasSig = true; document.getElementById('sigHint').style.display = 'none'; });

canvas.addEventListener('mouseup', () => drawing = false);

canvas.addEventListener('mouseleave', () => drawing = false);


canvas.addEventListener('touchstart', e => { e.preventDefault(); drawing = true; const p = getPos(e); ctx.beginPath(); ctx.moveTo(p.x, p.y); }, { passive: false });

canvas.addEventListener('touchmove', e => { e.preventDefault(); if (!drawing) return; const p = getPos(e); ctx.lineWidth = 2.5; ctx.lineCap = 'round'; ctx.strokeStyle = '#0A1628'; ctx.lineTo(p.x, p.y); ctx.stroke(); hasSig = true; document.getElementById('sigHint').style.display = 'none'; }, { passive: false });

canvas.addEventListener('touchend', () => drawing = false);

async function submitForm() {

  if (!validateStep(4)) return;


  const nomVal = parseFloat(document.getElementById('nominal').value);

  const sigData = canvas.toDataURL('image/png');

  const divisi = document.querySelector('input[name="divisi"]:checked').value;


  // convert files to base64

  const filesB64 = await Promise.all(uploadedFiles.map(f => new Promise(res => {

    const r = new FileReader(); r.onload = e => res(e.target.result); r.readAsDataURL(f);

  })));


  const payload = {

    timestamp: new Date().toLocaleString('id-ID'),

    nama: document.getElementById('nama').value.trim(),

    tanggal: document.getElementById('tanggal').value,

    noHP: document.getElementById('noHP').value.trim(),

    divisi,

    jenisKategori: document.getElementById('jenisKategori').value,

    nominal: nomVal,

    nominalFormatted: 'Rp ' + nomVal.toLocaleString('id-ID'),

    tujuan: document.getElementById('tujuan').value.trim(),

    keterangan: document.getElementById('keterangan').value.trim(),

    jumlahBukti: uploadedFiles.length,

    ttd: sigData,

    bukti: filesB64

  };


  if (!SCRIPT_URL) {

    // demo mode

    showSuccess(payload);

    showToast('⚠️ URL Sheet belum diset. Data tampil demo.');

    return;

  }


  document.getElementById('loadingOverlay').classList.add('show');

  try {

    const res = await fetch(SCRIPT_URL, {

      method: 'POST',

      mode: 'no-cors',

      headers: { 'Content-Type': 'application/json' },

      body: JSON.stringify(payload)

    });

    showSuccess(payload);

    showToast('✅ Data berhasil dikirim ke Google Sheets!');

  } catch (err) {

    showToast('❌ Gagal mengirim. Cek koneksi internet.');

  } finally {

    document.getElementById('loadingOverlay').classList.remove('show');

  }

}

3. Di baris 1, buat header mulai dari kolom B:

   B: Timestamp | C: Nama | D: Tanggal | E: No.HP | F: Divisi

   G: Kategori | H: Nominal | I: Tujuan | J: Keterangan | K: Jumlah Bukti


4. Buka menu Extensions > Apps Script

5. Hapus kode yang ada, ganti dengan kode berikut:


---

function doPost(e) {

  const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();

  const data = JSON.parse(e.postData.contents);


  sheet.appendRow([

    '',  // Kolom A kosong

    data.timestamp,

    data.nama,

    data.tanggal,

    data.noHP,

    data.divisi,

    data.jenisKategori,

    data.nominalFormatted,

    data.tujuan,

    data.keterangan || '-',

    data.jumlahBukti + ' file'

  ]);


  return ContentService

    .createTextOutput(JSON.stringify({status: 'ok'}))

    .setMimeType(ContentService.MimeType.JSON);

}

---


6. Save > Deploy > New Deployment

7. Type: Web App

8. Execute as: Me

9. Who has access: Anyone

10. Deploy > Copy URL

11. Paste URL ke form konfigurasi di bagian atas halaman
